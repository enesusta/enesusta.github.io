### 1

```java
public interface Animal {
    Optional<? extends Animal> getType();
    String getAnimalName();
}
```


```java
@Data
@AllArgsConstructor
public class DefaultAnimal implements Animal {

    private String name;

    @Override
    public Optional<? extends Animal> getType() {
        return Optional.of(this);
    }

    @Override
    public String getAnimalName() {
        return name;
    }

}
```

```java
@AllArgsConstructor
@Data
public class AnotherAnimal implements Animal {

    private String name;

    @Override
    public Optional<? extends Animal> getType() {
        return Optional.of(this);
    }

    @Override
    public String getAnimalName() {
        return name;
    }
}
```

```java
@Component
public class C08 implements Core {

    @Override
    public void run() {

        Animal defaultAnimal = new DefaultAnimal("essek");
        Animal another = new AnotherAnimal(null);

        String first = defaultAnimal
            .getType()
            .map(Animal::getAnimalName)
            .orElse("bilinmiyor");
        String sec = another
            .getType()
            .map(Animal::getAnimalName)
            .orElse("bilinmiyor2");

        System.out.println(first);
        System.out.println(sec);

    }

}
```

```
OUTPUT
=====
essek
bilinmiyor2
````

```java
@Component
public class C08 implements Core {

    @Override
    public void run() {

        Animal defaultAnimal = new DefaultAnimal("essek");
        Animal another = new AnotherAnimal(null);

        final List<Animal> animals = new ArrayList<>(2);
        animals.add(defaultAnimal);
        animals.add(another);

        printAllAnimalsName(animals);

    }

    private void printAllAnimalsName(final List<? extends Animal> list) {
        list
            .stream()
            .map(Animal::getAnimalName)
            .forEach(System.out::println);
    }

}
```
